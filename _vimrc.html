318 " 在保存文件时自动去除无效空白，包括行尾空白和文件最后的空行
319 " From: Vigil
320 " See: http://blog.bs2.to/post/EdwardLee/17961
321 function! RemoveTrailingWhitespace()
322     if &ft != "diff"
323         let b:curcol = col(".")
324         let b:curline = line(".")
325         silent! %s/\s\+$//
326         silent! %s/\(\s*\n\)\+\%$//
327         call cursor(b:curline, b:curcol)
328     endif
329 endfunction

812 function! SelectFuzzyFinderMode()
813     let fufmodelist = [
814                 \ { "FufBuffer"       : "Buffer mode (fuf-buffer-mode)" },
815                 \ { "FufFile"         : "File mode (fuf-file-mode)" },
816                 \ { "FufCoverageFile" : "Coverage-File mode (fuf-coveragefile-mode)" },
817                 \ { "FufDir"          : "Directory mode (fuf-dir-mode)" },
818                 \ { "FufMruFile"      : "MRU-File mode (fuf-mrufile-mode)" },
819                 \ { "FufMruCmd"       : "MRU-Command mode (fuf-mrucmd-mode)" },
820                 \ { "FufBookmarkFile" : "Bookmark-File mode (fuf-bookmarkfile-mode)" },
821                 \ { "FufBookmarkDir"  : "Bookmark-Dir mode (fuf-bookmarkdir-mode)" },
822                 \ { "FufTag"          : "Tag mode (fuf-tag-mode)" },
823                 \ { "FufBufferTag"    : "Buffer-Tag mode (fuf-buffertag-mode)" },
824                 \ { "FufTaggedFile"   : "Tagged-File mode (fuf-taggedfile-mode)" },
825                 \ { "FufJumpList"     : "Jump-List mode (fuf-jumplist-mode)" },
826                 \ { "FufChangeList"   : "Change-List mode (fuf-changelist-mode)" },
827                 \ { "FufQuickfix"     : "Quickfix mode (fuf-quickfix-mode)" },
828                 \ { "FufLine"         : "Line mode (fuf-line-mode)" },
829                 \ { "FufHelp"         : "Help mode (fuf-help-mode)" }
830                 \ ]
831     let fufmodenum = 1
832     let fufmodecmddic = {}
833     let fufmodeliststr = [printf("%4s %-16s\t%s","序号","命令","说明")]
834     for t1 in fufmodelist
835         let tl1 = items(t1)[0]
836         let fufmodecmddic[fufmodenum] = tl1[0]
837         let fufmodestrlist = add(fufmodeliststr ,  printf("%4d %-16s\t%s",fufmodenum,tl1[0],tl1[1]))
838         let fufmodenum = fufmodenum + 1
839     endfor
840     redraw
841     let selectmode = inputlist(fufmodestrlist)
842     if selectmode<1 || selectmode>len(fufmodestrlist)
843         return
844     endif
845     exe fufmodecmddic[selectmode]
846 endfunction
