292 " 比较函数
293 function! MyDiff()
294     let opt = '-a --binary '
295     if &diffopt =~ 'icase' | let opt = opt . '-i ' | endif
296     if &diffopt =~ 'iwhite' | let opt = opt . '-b ' | endif
297     let arg1 = v:fname_in
298     if arg1 =~ ' ' | let arg1 = '"' . arg1 . '"' | endif
299     let arg2 = v:fname_new
300     if arg2 =~ ' ' | let arg2 = '"' . arg2 . '"' | endif
301     let arg3 = v:fname_out
302     if arg3 =~ ' ' | let arg3 = '"' . arg3 . '"' | endif
303     let eq = ''
304     if $VIMRUNTIME =~ ' '
305         if &sh =~ '\<cmd'
306             let cmd = '""' . $VIMRUNTIME . '\diff"'
307             let eq = '"'
308         else
309             let cmd = substitute($VIMRUNTIME, ' ', '" ', '') . '\diff"'
310         endif
311     else
312         let cmd = $VIMRUNTIME . '\diff'
313     endif
314     silent execute '!' . cmd . ' ' . opt . arg1 . ' ' . arg2 . ' > ' . arg3 . eq
315 endfunction
316 if MySys()=="windows"
317     set diffexpr=MyDiff()
318 endif

320 " 在保存文件时自动去除无效空白，包括行尾空白和文件最后的空行
321 " From: Vigil
322 " See: http://blog.bs2.to/post/EdwardLee/17961
323 function! RemoveTrailingWhitespace()
324     if &ft != "diff"
325         let b:curcol = col(".")
326         let b:curline = line(".")
327         silent! %s/\s\+$//
328         silent! %s/\(\s*\n\)\+\%$//
329         call cursor(b:curline, b:curcol)
330     endif
331 endfunction
332 autocmd BufWritePre * call RemoveTrailingWhitespace()
