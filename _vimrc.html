 335 " 在保存文件时自动去除无效空白，包括行尾空白和文件最后的空行
 336 " From: Vigil
 337 " See: http://blog.bs2.to/post/EdwardLee/17961
 338 function! RemoveTrailingWhitespace()
 339     if &ft != "diff"
 340         let b:curcol = col(".")
 341         let b:curline = line(".")
 342         silent! %s/\s\+$//
 343         silent! %s/\(\s*\n\)\+\%$//
 344         call cursor(b:curline, b:curcol)
 345     endif
 346 endfunction

 808 function! SelectFuzzyFinderMode()
 809     let fufmodelist = [
 810                 \ { "FufBuffer"       : "Buffer mode (fuf-buffer-mode)" },
 811                 \ { "FufFile"         : "File mode (fuf-file-mode)" },
 812                 \ { "FufCoverageFile" : "Coverage-File mode (fuf-coveragefile-mode)" },
 813                 \ { "FufDir"          : "Directory mode (fuf-dir-mode)" },
 814                 \ { "FufMruFile"      : "MRU-File mode (fuf-mrufile-mode)" },
 815                 \ { "FufMruCmd"       : "MRU-Command mode (fuf-mrucmd-mode)" },
 816                 \ { "FufBookmarkFile" : "Bookmark-File mode (fuf-bookmarkfile-mode)" },
 817                 \ { "FufBookmarkDir"  : "Bookmark-Dir mode (fuf-bookmarkdir-mode)" },
 818                 \ { "FufTag"          : "Tag mode (fuf-tag-mode)" },
 819                 \ { "FufBufferTag"    : "Buffer-Tag mode (fuf-buffertag-mode)" },
 820                 \ { "FufTaggedFile"   : "Tagged-File mode (fuf-taggedfile-mode)" },
 821                 \ { "FufJumpList"     : "Jump-List mode (fuf-jumplist-mode)" },
 822                 \ { "FufChangeList"   : "Change-List mode (fuf-changelist-mode)" },
 823                 \ { "FufQuickfix"     : "Quickfix mode (fuf-quickfix-mode)" },
 824                 \ { "FufLine"         : "Line mode (fuf-line-mode)" },
 825                 \ { "FufHelp"         : "Help mode (fuf-help-mode)" }
 826                 \ ]
 827     let fufmodenum = 1
 828     let fufmodecmddic = {}
 829     let fufmodeliststr = [printf("%4s %-16s\t%s","序号","命令","说明")]
 830     for t1 in fufmodelist
 831         let tl1 = items(t1)[0]
 832         let fufmodecmddic[fufmodenum] = tl1[0]
 833         let fufmodestrlist = add(fufmodeliststr ,  printf("%4d %-16s\t%s",fufmodenum,tl1[0],tl1[1]))
 834         let fufmodenum = fufmodenum + 1
 835     endfor
 836     redraw
 837     let selectmode = inputlist(fufmodestrlist)
 838     if selectmode<1 || selectmode>len(fufmodestrlist)
 839         return
 840     endif
 841     exe fufmodecmddic[selectmode]
 842 endfunction
